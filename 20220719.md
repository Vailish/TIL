# 구분자.join(리스트) : 구분자 = '' ""

word = list(input())
k = 0
for n in range(len(word)):
    if word[n] == 'a':
        k = k + 1
while k != 0:
    word.remove('a')
    k = k - 1

print(''.join(word))
